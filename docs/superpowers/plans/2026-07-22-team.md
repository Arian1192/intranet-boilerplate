```ts
// src/features/team/data/absences.ts
import { allPeople, type Person } from './people';

// tabla absences (espejo Supabase) — Calendario del equipo
export type AbsenceType = 'vacaciones' | 'teletrabajo' | 'baja' | 'ausencia';

export interface Absence {
  id: string;
  personId: string;
  type: AbsenceType;
  startDate: string; // ISO 'YYYY-MM-DD'
  endDate: string; // ISO 'YYYY-MM-DD' (inclusive)
  approved: boolean;
}

export const absences: Absence[] = [
  { id: 'ab-alejandro-gonzalez', personId: 'alejandro-gonzalez', type: 'vacaciones', startDate: '2026-07-03', endDate: '2026-07-07', approved: true },
  { id: 'ab-carlos-pego-munoz', personId: 'carlos-pego-munoz', type: 'vacaciones', startDate: '2026-07-03', endDate: '2026-07-05', approved: true },
  { id: 'ab-carlos-ramudo-valencia', personId: 'carlos-ramudo-valencia', type: 'vacaciones', startDate: '2026-07-16', endDate: '2026-07-22', approved: true },
];

function parseISODate(iso: string): Date {
  const [y, m, d] = iso.split('-').map(Number);
  return new Date(y, m - 1, d);
}

export function absencesForMonth(
  year: number,
  month: number,
  opts?: { approvedOnly?: boolean }
): Absence[] {
  const monthStart = new Date(year, month, 1);
  const monthEnd = new Date(year, month + 1, 0);
  return absences.filter((a) => {
    if (opts?.approvedOnly && !a.approved) return false;
    const start = parseISODate(a.startDate);
    const end = parseISODate(a.endDate);
    return start <= monthEnd && end >= monthStart;
  });
}

export function absenceDays(absence: Absence): number {
  const start = parseISODate(absence.startDate);
  const end = parseISODate(absence.endDate);
  const diffMs = end.getTime() - start.getTime();
  return Math.round(diffMs / 86400000) + 1;
}

export function todaysAbsence(isoToday: string): { person: Person; absence: Absence } | undefined {
  const today = parseISODate(isoToday);
  const hit = absences.find((a) => {
    const start = parseISODate(a.startDate);
    const end = parseISODate(a.endDate);
    return today >= start && today <= end;
  });
  if (!hit) return undefined;
  const person = allPeople().find((p) => p.id === hit.personId);
  if (!person) return undefined;
  return { person, absence: hit };
}

export function totalDaysForPerson(personId: string, year: number, month: number): number | undefined {
  const list = absencesForMonth(year, month).filter((a) => a.personId === personId);
  if (list.length === 0) return undefined;
  return list.reduce((sum, a) => sum + absenceDays(a), 0);
}
```

```ts
  type?: Ficha['employmentType'] | 'todos';

  return list.filter((item) => {
    if (f.companyId && f.companyId !== 'todas' && !item.companyIds.includes(f.companyId)) return false;

export function costRankingByPerson(): { person: Person; ficha: Ficha }[] {

  return fichas
```

```tsx
import { cn } from '@/lib/utils';

export interface SegmentedControlOption<T extends string> {
  label: string;
  value: T;
}

export interface SegmentedControlProps<T extends string> {
  options: SegmentedControlOption<T>[];
  value: T;
  onChange: (value: T) => void;
  fullWidth?: boolean;
  className?: string;
  tone?: 'light' | 'dark';
}

export function SegmentedControl<T extends string>({
  options,
  value,
  onChange,
  fullWidth = false,
  className,
  tone = 'light',
}: SegmentedControlProps<T>) {
  return (
    <div
      className={cn(
        'rounded-lg bg-slate-100 p-1',
        fullWidth ? 'grid w-full auto-cols-fr grid-flow-col gap-1' : 'inline-flex items-center gap-1',
        className
      )}
    >
      {options.map((option) => {
        const active = value === option.value;
        return (
          <button
            key={option.value}
            type="button"
            onClick={() => onChange(option.value)}
            className={cn(
              'rounded-md px-3 py-1 text-sm font-medium transition-colors',
              active
                ? tone === 'dark'
                  ? 'text-white shadow-sm'
                  : 'bg-white text-slate-800 shadow-sm'
                : 'text-slate-500 hover:text-slate-700'
            )}
            style={active && tone === 'dark' ? { backgroundColor: '#44444C' } : undefined}
          >
            {option.label}
          </button>
        );
      })}
    </div>
  );
}
```

```tsx
// src/features/team/pages/EquipoPage.tsx
import { useMemo, useState } from 'react';
import { SegmentedControl } from '@/components/ui';
import { DirectorioToolbar } from '../components/DirectorioToolbar';
import { PersonCard } from '../components/PersonCard';
import { OrgRow } from '../components/OrgRow';
import { allPeople, searchPeople, departmentsList, directReports, orgRoots, type Person } from '../data/people';

type View = 'directorio' | 'organigrama';

function renderOrgTree(person: Person, level: number): JSX.Element {
  const reports = directReports(person.id);
  return (
    <div key={person.id}>
      <OrgRow person={person} reportsCount={reports.length} level={level} />
      {reports.map((r) => renderOrgTree(r, level + 1))}
    </div>
  );
}

export function EquipoPage() {
  const [view, setView] = useState<View>('directorio');
  const [query, setQuery] = useState('');
  const [department, setDepartment] = useState('');

  const people = useMemo(() => allPeople(), []);
  const departments = useMemo(() => departmentsList(), []);
  const visible = useMemo(
    () => searchPeople(people, query, department || undefined),
    [people, query, department]
  );

  function managerName(person: Person): string | undefined {
    return person.managerId ? people.find((p) => p.id === person.managerId)?.name : undefined;
  }

  return (
    <div className="space-y-4">
      <div>
        <h1 className="text-2xl font-semibold text-slate-800">Equipo</h1>
        <p className="text-sm text-slate-500">Quién es quién, cómo contactarle y de quién depende.</p>
      </div>

      <SegmentedControl
        tone="dark"
        options={[
          { label: 'Directorio', value: 'directorio' },
          { label: 'Organigrama', value: 'organigrama' },
        ]}
        value={view}
        onChange={setView}
      />

      {view === 'directorio' ? (
        <div className="space-y-4">
          <DirectorioToolbar
            query={query}
            onQuery={setQuery}
            department={department}
            onDepartment={setDepartment}
            departments={departments}
          />
          <div className="grid grid-cols-1 gap-4 sm:grid-cols-2 lg:grid-cols-3">
            {visible.map((person) => (
              <PersonCard key={person.id} person={person} managerName={managerName(person)} />
            ))}
          </div>
        </div>
      ) : (
        <div className="rounded-xl border border-slate-100 bg-white p-2 shadow-sm">
          {orgRoots().map((root) => renderOrgTree(root, 0))}
        </div>
      )}
    </div>
  );
}
```

```tsx
// src/features/team/components/TeamCalendarGrid.tsx
import { cn } from '@/lib/utils';
import { Avatar } from '@/components/ui';
import { absenceDays, type Absence, type AbsenceType } from '../data/absences';
import type { Person } from '../data/people';

const TYPE_COLOR: Record<AbsenceType, string> = {
  vacaciones: '#7c3aed',
  teletrabajo: '#2563eb',
  baja: '#dc2626',
  ausencia: '#ea580c',
};

// getDay(): 0=Domingo … 6=Sábado; letras del live (X/J/V/S/D/L/M)
const WEEKDAY_LETTERS = ['D', 'L', 'M', 'X', 'J', 'V', 'S'];

export interface TeamCalendarGridProps {
  people: Person[];
  year: number;
  month: number; // 0-11
  absences: Absence[]; // ya filtradas por mes y por "Solo aprobadas" si aplica
  todayIso: string; // 'YYYY-MM-DD'
}

export function TeamCalendarGrid({ people, year, month, absences, todayIso }: TeamCalendarGridProps) {
  const daysInMonth = new Date(year, month + 1, 0).getDate();
  const days = Array.from({ length: daysInMonth }, (_, i) => i + 1);
  const [todayY, todayM, todayD] = todayIso.split('-').map(Number);
  const isTodayMonth = todayY === year && todayM - 1 === month;

  const absencesByPerson = new Map<string, Absence[]>();
  absences.forEach((a) => {
    const list = absencesByPerson.get(a.personId) ?? [];
    list.push(a);
    absencesByPerson.set(a.personId, list);
  });

  function absenceOnDay(personAbsences: Absence[], day: number): Absence | undefined {
    const iso = `${year}-${String(month + 1).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
    return personAbsences.find((a) => a.startDate <= iso && iso <= a.endDate);
  }

  return (
    <div className="overflow-x-auto rounded-xl border border-slate-100 bg-white shadow-sm">
      <table className="w-full border-collapse text-xs">
        <thead>
          <tr>
            <th className="sticky left-0 bg-white px-3 py-2 text-left font-medium text-slate-500">Persona</th>
            {days.map((day) => {
              const weekday = new Date(year, month, day).getDay();
              const weekend = weekday === 0 || weekday === 6;
              const isToday = isTodayMonth && day === todayD;
              return (
                <th
                  key={day}
                  data-testid={`day-header-${day}`}
                  className={cn(
                    'px-1 py-2 text-center font-medium text-slate-400',
                    weekend && 'bg-slate-50',
                    isToday && 'bg-slate-200 text-slate-700'
                  )}
                >
                  <div>{WEEKDAY_LETTERS[weekday]}</div>
                  <div>{day}</div>
                </th>
              );
            })}
            <th className="px-3 py-2 text-right font-medium text-slate-500">Días</th>
          </tr>
        </thead>
        <tbody>
          {people.map((person) => {
            const personAbsences = absencesByPerson.get(person.id) ?? [];
            const totalDays = personAbsences.reduce((sum, a) => sum + absenceDays(a), 0);
            return (
              <tr
                key={person.id}
                data-testid={`calendar-row-${person.id}`}
                className="border-t border-slate-100 hover:bg-slate-50"
              >
                <td className="sticky left-0 flex items-center gap-2 bg-white px-3 py-2">
                  <Avatar
                    src={person.photoUrl}
                    alt={person.name}
                    fallback={person.name}
                    size="sm"
                    style={{ backgroundColor: person.avatarColor }}
                  />
                  <span className="whitespace-nowrap font-medium text-slate-700">{person.name}</span>
                </td>
                {days.map((day) => {
                  const weekday = new Date(year, month, day).getDay();
                  const weekend = weekday === 0 || weekday === 6;
                  const isToday = isTodayMonth && day === todayD;
                  const absence = absenceOnDay(personAbsences, day);
                  return (
                    <td
                      key={day}
                      data-testid={`cell-${person.id}-${day}`}
                      className={cn('relative h-8 px-0.5', weekend && 'bg-slate-50', isToday && 'bg-slate-100')}
                    >
                      {absence && (
                        <div
                          data-testid={`bar-${person.id}-${day}`}
                          className={cn('h-4 w-full', absence.approved ? 'opacity-100' : 'opacity-40')}
                          style={{ backgroundColor: TYPE_COLOR[absence.type] }}
                        />
                      )}
                    </td>
                  );
                })}
                <td className="px-3 py-2 text-right font-medium text-slate-600">
                  {totalDays > 0 ? totalDays : '—'}
                </td>
              </tr>
            );
          })}
        </tbody>
      </table>
    </div>
  );
}
```

```tsx
import { useMemo, useState } from 'react';
import { TeamCalendarGrid } from '../components/TeamCalendarGrid';
import { CalendarLegend } from '../components/CalendarLegend';
import { allPeople } from '../data/people';
import { absencesForMonth, todaysAbsence, type AbsenceType } from '../data/absences';

const TODAY_ISO = '2026-07-22';
const MONTH_LABELS = [
  'Enero', 'Febrero', 'Marzo', 'Abril', 'Mayo', 'Junio',
  'Julio', 'Agosto', 'Septiembre', 'Octubre', 'Noviembre', 'Diciembre',
];
const TYPE_LABEL: Record<AbsenceType, string> = {
  vacaciones: 'Vacaciones',
  teletrabajo: 'Teletrabajo',
  baja: 'Baja',
  ausencia: 'Ausencia',
};

export function CalendarioPage() {
  const [year, setYear] = useState(2026);
  const [month, setMonth] = useState(6); // julio, 0-indexado
  const [approvedOnly, setApprovedOnly] = useState(true);

  const people = useMemo(() => allPeople(), []);
  const absences = useMemo(
    () => absencesForMonth(year, month, { approvedOnly }),
    [year, month, approvedOnly]
  );
  const today = todaysAbsence(TODAY_ISO);

  function prevMonth() {
    if (month === 0) {
      setYear((y) => y - 1);
      setMonth(11);
    } else {
      setMonth((m) => m - 1);
    }
  }

  function nextMonth() {
    if (month === 11) {
      setYear((y) => y + 1);
      setMonth(0);
    } else {
      setMonth((m) => m + 1);
    }
  }

  return (
    <div className="space-y-4">
      <div>
        <h1 className="text-2xl font-semibold text-slate-800">Calendario del equipo</h1>

      </div>

      <div className="flex items-center gap-3">
        <button type="button" onClick={prevMonth} aria-label="Mes anterior" className="text-slate-400 hover:text-slate-600">
          ‹
        </button>
        <span className="text-sm font-medium text-slate-700">{MONTH_LABELS[month]} {year}</span>
        <button type="button" onClick={nextMonth} aria-label="Mes siguiente" className="text-slate-400 hover:text-slate-600">
          ›
        </button>
      </div>

        <p className="text-sm text-slate-600">
```

```tsx
// src/components/ui/ProgressBar.tsx
import { cn } from '@/lib/utils';

export interface ProgressBarProps {
  value: number;
  max: number;
  className?: string;
  fillClassName?: string;
  fillStyle?: React.CSSProperties;
}

export function ProgressBar({ value, max, className, fillClassName, fillStyle }: ProgressBarProps) {
  const percent = max > 0 ? Math.min(100, Math.max(0, (value / max) * 100)) : 0;
  return (
    <div
      role="progressbar"
      aria-valuemin={0}
      aria-valuenow={value}
      aria-valuemax={max}
      className={cn('h-2 w-full overflow-hidden rounded-full bg-slate-200', className)}
    >
      <div
        className={cn('h-full rounded-full', !fillStyle && (fillClassName ?? 'bg-emerald-500'))}
        style={{ width: `${percent}%`, ...fillStyle }}
      />
    </div>
  );
}
```

```tsx
// src/components/ui/MasterDetailList.tsx
import { useState } from 'react';
import { Card } from './Card';
import { cn } from '@/lib/utils';

export interface MasterDetailListProps<T extends { id: string }> {
  items: T[];
  renderRow: (item: T, isSelected: boolean) => React.ReactNode;
  renderDetail: (item: T) => React.ReactNode;
  emptyState?: string;
  className?: string;
  selectedId?: string | null;
  onSelect?: (id: string) => void;
  listTop?: React.ReactNode;
  detailOverride?: React.ReactNode;
  selectedRowClassName?: string;
}

export function MasterDetailList<T extends { id: string }>({
  items,
  renderRow,
  renderDetail,
  emptyState = 'Selecciona un elemento o crea uno nuevo.',
  className,
  selectedId: controlledSelectedId,
  onSelect,
  listTop,
  detailOverride,
  selectedRowClassName = 'bg-brand-50/60',
}: MasterDetailListProps<T>) {
  const [internalSelectedId, setInternalSelectedId] = useState<string | null>(null);
  const selectedId = controlledSelectedId !== undefined ? controlledSelectedId : internalSelectedId;
  const selected = items.find((item) => item.id === selectedId) ?? null;

  const handleSelect = (id: string) => {
    onSelect?.(id);
    if (controlledSelectedId === undefined) setInternalSelectedId(id);
  };

  return (
    <div className={cn('grid items-start gap-6 lg:grid-cols-[400px_1fr]', className)}>
      <div className="space-y-4">
        {listTop}
        <Card className="divide-y divide-slate-100 p-0">
          {items.map((item) => (
            <button
              key={item.id}
              type="button"
              onClick={() => handleSelect(item.id)}
              className={cn(
                'block w-full px-4 py-3 text-left transition-colors hover:bg-slate-50',
                selectedId === item.id && selectedRowClassName
              )}
            >
              {renderRow(item, selectedId === item.id)}
            </button>
          ))}
        </Card>
      </div>
      {detailOverride ? (
        <Card className="p-6">{detailOverride}</Card>
      ) : selected ? (
        <Card className="p-6">{renderDetail(selected)}</Card>
      ) : (
        <Card className="flex min-h-[200px] items-center justify-center p-6">
          <p className="text-slate-400">{emptyState}</p>
        </Card>
      )}
    </div>
  );
}
```

```tsx
// src/features/team/pages/FichasPage.tsx
import { useMemo } from 'react';
import { useSearchParams } from 'react-router';
import { Button, MasterDetailList } from '@/components/ui';
import { allPeople, type Person } from '../data/people';
import {
  fichas as allFichas, filterFichas, fichaFor, companies, totalMonthlyCost, costRankingByPerson,
  type CompanyId, type Ficha,
} from '../data/fichas';
import { FichasFilterPanel, type CompanyFilterValue, type TypeFilterValue } from '../components/FichasFilterPanel';
import { FichaListRow } from '../components/FichaListRow';
import { FichaDetailPanel } from '../components/FichaDetailPanel';
import { CostDashboard } from '../components/CostDashboard';

interface FichaRow {
  id: string; // person.id
  person: Person;
  ficha: Ficha;
}

export function FichasPage() {
  const [searchParams, setSearchParams] = useSearchParams();
  const selectedPersonId = searchParams.get('p');
  const companyId = (searchParams.get('empresa') as CompanyFilterValue) || 'todas';
  const type = (searchParams.get('tipo') as TypeFilterValue) || 'todos';

  const people = useMemo(() => allPeople(), []);

  const rows: FichaRow[] = useMemo(() => {
    const visible = filterFichas(allFichas, {
      companyId: companyId === 'todas' ? undefined : (companyId as CompanyId),
      type: type === 'todos' ? undefined : type,
    });
    return visible
      .map((ficha) => {
        const person = people.find((p) => p.id === ficha.personId);
        return person ? { id: person.id, person, ficha } : undefined;
      })
      .filter((r): r is FichaRow => r !== undefined);
  }, [people, companyId, type]);

  function updateParam(key: string, value: string | undefined) {
    setSearchParams((prev) => {
      const next = new URLSearchParams(prev);
      if (value) next.set(key, value); else next.delete(key);
      return next;
    });
  }

  const selectedPerson = selectedPersonId ? people.find((p) => p.id === selectedPersonId) : undefined;
  const selectedFicha = selectedPersonId ? fichaFor(selectedPersonId) : undefined;

  return (
    <div className="space-y-6">
      <div className="flex items-start justify-between gap-4">
        <div>
          <h1 className="text-2xl font-semibold text-slate-800">Fichas del equipo</h1>
          <p className="text-sm text-slate-500">
            Información sensible: datos personales, condiciones económicas, vacaciones y horas. El resto del
            equipo solo ve el directorio.
          </p>
        </div>
        <Button type="button" variant="secondary">⬇ Exportar a Excel</Button>
      </div>

      <MasterDetailList<FichaRow>
        items={rows}
        renderRow={(row) => <FichaListRow person={row.person} ficha={row.ficha} />}
        renderDetail={() => null}
        selectedId={selectedPersonId}
        onSelect={(id) => updateParam('p', id)}
        selectedRowClassName="bg-slate-100"
        emptyState="Selecciona una persona o crea una nueva."
        listTop={
          <FichasFilterPanel
            companyId={companyId}
            onCompanyId={(v) => updateParam('empresa', v === 'todas' ? undefined : v)}
            type={type}
            onType={(v) => updateParam('tipo', v === 'todos' ? undefined : v)}
          />
        }
        detailOverride={
          selectedPerson && selectedFicha ? (
            <FichaDetailPanel person={selectedPerson} ficha={selectedFicha} allPeople={people} />
          ) : undefined
        }
      />

      <CostDashboard total={totalMonthlyCost()} companies={companies()} ranking={costRankingByPerson()} />
    </div>
  );
}
```
